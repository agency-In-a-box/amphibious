Nested Grids

You can nest grids within columns for complex layouts:

Header
Navigation
Main Content
Sidebar
Footer
<div class="container">
  <!-- Header -->
  <div class="row">
    <header class="col-16">Header</header>
  </div>

  <!-- Content Row -->
  <div class="row">
    <!-- Navigation -->
    <nav class="col-5">Navigation</nav>

    <!-- Main + Sidebar nested grid -->
    <div class="col-11">
      <div class="row">
        <main class="col-12">Main Content</main>
        <aside class="col-4">Sidebar</aside>
      </div>
    </div>
  </div>

  <!-- Footer -->
  <div class="row">
    <footer class="col-16">Footer</footer>
  </div>
</div>
