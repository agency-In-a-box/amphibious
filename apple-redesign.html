Quick Start

# Install Amphibious 2.0
npm install @amphibious/css

# Or using CDN
<link rel="stylesheet" href="https://unpkg.com/@amphibious/css@2.0/dist/apple.css">

Basic HTML Template

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>My Amphibious App</title>
    <link rel="stylesheet" href="amphibious/apple.css">
</head>
<body>
    <!-- Navigation -->
    <nav class="apple-nav">
        <div class="apple-nav-container">
            <a href="#" class="apple-nav-logo">Your Logo</a>
            <ul class="apple-nav-menu">
                <li><a href="#" class="apple-nav-item">Home</a></li>
                <li><a href="#" class="apple-nav-item">About</a></li>
            </ul>
        </div>
    </nav>

    <!-- Hero Section -->
    <section class="apple-hero-section">
        <h1 class="apple-hero-title">Welcome</h1>
        <p class="apple-hero-subtitle">Build something amazing</p>
        <button class="apple-button">Get Started</button>
    </section>
</body>
</html>
